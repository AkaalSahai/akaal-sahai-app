-- Add permission flag to users table
ALTER TABLE public.users ADD COLUMN IF NOT EXISTS can_edit_students boolean DEFAULT false;

-- Drop the open teacher policies (if you ran them earlier)
DROP POLICY IF EXISTS "students_teacher_insert" ON public.students;
DROP POLICY IF EXISTS "students_teacher_update" ON public.students;

-- Recreate with permission gate
CREATE POLICY "students_teacher_insert" ON public.students FOR INSERT WITH CHECK (
  user_has_role('teacher')
  AND (SELECT can_edit_students FROM public.users WHERE id = auth.uid()) = true
  AND group_id IN (SELECT id FROM public.groups WHERE teacher_id = auth.uid())
);

CREATE POLICY "students_teacher_update" ON public.students FOR UPDATE USING (
  user_has_role('teacher')
  AND (SELECT can_edit_students FROM public.users WHERE id = auth.uid()) = true
  AND group_id IN (SELECT id FROM public.groups WHERE teacher_id = auth.uid())
);
